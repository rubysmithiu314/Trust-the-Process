${p.name}
${p.role}
`; }).join(''); } // ============================================================ // GAME LOGIC CORE // ============================================================ function startGame() { const room = getRoom(myCode); if (!room) return; Object.keys(room.players).forEach(pName => { const p = room.players[pName]; if (p.role === 'NP') { p.hand = [room.npPile.pop(), room.npPile.pop(), room.npPile.pop()]; } else if (p.role === 'FA' || p.role === 'BANK') { p.hand = [room.faPile.pop(), room.faPile.pop()]; } else if (p.role === 'FB') { p.hand = [room.fbPile.pop(), room.fbPile.pop()]; } }); room.status = 'playing'; room.log.push('Game started.'); saveRoom(myCode, room); enterGame(); } function enterGame() { if (myRole === 'NP') { document.getElementById('npName').innerText = myName; showScreen('screenGameNP'); } else { document.getElementById('fName').innerText = myName; document.getElementById('fRoleLabel').innerText = myRole; showScreen('screenGameF'); if (myRole === 'BANK') { document.getElementById('bankerSection').style.display = 'block'; } } } function renderGame(room) { const me = room.players[myName]; if (!me) return; if (myRole === 'NP') { document.getElementById('npTokens').innerText = me.tokens; document.getElementById('handCount').innerText = `(${me.hand.length})`; const handDiv = document.getElementById('npHand'); handDiv.innerHTML = me.hand.map(cKey => { const c = CD[cKey]; return `
${c.icon}
${c.name}
`; }).join(''); const applyHandDiv = document.getElementById('npApplyHand'); applyHandDiv.innerHTML = me.hand.map((cKey, idx) => { const c = CD[cKey]; const isSel = selApply.includes(idx); return `
${c.icon}
${c.name}
`; }).join(''); const selCardsDiv = document.getElementById('selectedCards'); selCardsDiv.innerText = selApply.length === 0 ? CD[fA.lockedProgram].name : 'Not locked yet'; if (fB) { document.getElementById('npFBProgram').innerText = fB.lockedProgram ? CD[fB.lockedProgram].name : 'Not locked yet'; const req = trustReq(me.oakApprovals); document.getElementById('oakReqLabel').innerText = `1 prog + ${req.actions} action + ${req.ciu} CIU`; } document.getElementById('npGameLog').innerHTML = room.log.slice(-10).reverse().map(l => `
• ${l}
`).join(''); document.getElementById('gainTokenBtn').classList.toggle('btn-off', me.gainedCIU); document.getElementById('drawCardBtn').classList.toggle('btn-off', me.drewCard); } else { // Foundation side document.getElementById('fFundingLeft').innerText = (myRole === 'FB') ? (3 - room.fundingGiven.FB) : (3 - room.fundingGiven.FA); const handDiv = document.getElementById('fHand'); handDiv.innerHTML = me.hand.map(cKey => { const c = CD[cKey]; return `

`).join(''); } document.getElementById('fGameLog').innerHTML = room.log.slice(-10).reverse().map(l => `
• ${l}
`).join(''); } } // ============================================================ // INTERACTIONS // ============================================================ function toggleApply(idx) { const i = selApply.indexOf(idx); if (i > -1) selApply.splice(i, 1); else selApply.push(idx); renderGame(getRoom(myCode)); } function gainToken() { const room = getRoom(myCode); const me = room.players[myName]; if (me.gainedCIU) return; me.tokens++; me.gainedCIU = true; room.log.push(`${myName} gained 1 CIU.`); saveRoom(myCode, room); renderGame(room); } function drawCard() { const room = getRoom(myCode); const me = room.players[myName]; if (me.drewCard || room.npPile.length === 0) return; me.hand.push(room.npPile.pop()); me.drewCard = true; me.gainedCIU = false; me.drewCard = false; // Reset for next turn room.log.push(`${myName} drew a card.`); saveRoom(myCode, room); renderGame(room); } function fDrawCard() { const room = getRoom(myCode); const me = room.players[myName]; const pile = (myRole === 'FB') ? room.fbPile : room.faPile; if (pile.length === 0) return; me.hand.push(pile.pop()); room.log.push(`${myName} drew a card.`); saveRoom(myCode, room); renderGame(room); } function playProgramCard() { const room = getRoom(myCode); const me = room.players[myName]; const progIdx = me.hand.findIndex(c => CD[c].type === 'program'); if (progIdx > -1) { me.lockedProgram = me.hand[progIdx]; me.hand.splice(progIdx, 1); room.log.push(`${myName} locked in program ${CD[me.lockedProgram].name}.`); saveRoom(myCode, room); renderGame(room); } } function applyToFoundation(funder) { const room = getRoom(myCode); const me = room.players[myName]; const selectedCards = selApply.map(idx => me.hand[idx]); const programs = selectedCards.filter(c => CD[c].type === 'program'); const actions = selectedCards.filter(c => CD[c].type === 'action'); const req = (funder === 'FB') ? trustReq(me.oakApprovals) : { actions: 3, ciu: 3 }; if (programs.length !== 1 || actions.length < req.actions || me.tokens < req.ciu) { notify('Missing required cards or CIU!', 'error'); return; } room.applications.push({ id: Math.random().toString(36).substr(2, 5), from: myName, to: funder, cards: selectedCards, status: 'pending' }); me.hand = me.hand.filter((_, idx) => !selApply.includes(idx)); me.tokens -= req.ciu; selApply = []; room.log.push(`${myName} applied to ${funder}.`); saveRoom(myCode, room); notify('Application submitted', 'success'); switchTab('hand'); } function approveApp(appId) { const room = getRoom(myCode); const app = room.applications.find(a => a.id === appId); const applicant = room.players[app.from]; app.status = 'approved'; if (myRole === 'FB') { room.fundingGiven.FB++; applicant.oakApprovals++; applicant.hand.push('unrestricted'); } else { room.fundingGiven.FA++; applicant.hand.push('restricted'); } room.log.push(`${myName} approved ${app.from}.`); saveRoom(myCode, room); } function denyApp(appId) { const room = getRoom(myCode); const app = room.applications.find(a => a.id === appId); app.status = 'denied'; room.log.push(`${myName} denied ${app.from}.`); saveRoom(myCode, room); } // ============================================================ // UI NAVIGATION & HELPERS // ============================================================ function switchTab(t) { document.querySelectorAll('.tc').forEach(c => c.classList.remove('active')); document.querySelectorAll('.tbtn').forEach(b => b.classList.remove('active')); document.getElementById('tabContent' + t.charAt(0).toUpperCase() + t.slice(1)).classList.add('active'); document.getElementById('tab' + t.charAt(0).toUpperCase() + t.slice(1)).classList.add('active'); } function switchFTab(t) { document.querySelectorAll('#screenGameF .tc').forEach(c => c.classList.remove('active')); document.querySelectorAll('#screenGameF .tbtn').forEach(b => b.classList.remove('active')); const n = document.createElement('div'); n.className = `notif ${type}`; n.innerText = msg; area.appendChild(n); setTimeout(() => n.remove(), 3000); } const savedMe = localStorage.getItem('ttp_me'); if (savedMe) { const data = JSON.parse(savedMe);
